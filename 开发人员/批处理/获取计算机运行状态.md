# 获取计算机运行状态.
```
 for /f "tokens=2 delims=" %%a in ('wmic path Win32_PerfFormattedData_PerfOS_Processor get PercentProcessorTime /value^|findstr "PercentProcessorTime"') do (set ucpu=%%a)  
 for /f "tokens=2 delims==" %%a in ('wmic ComputerSystem get TotalPhysicalMemory /value^|findstr "TotalPhysicalMemory"') do (set totalMem=%%a)
 for /f "tokens=2 delims==" %%a in ('wmic path Win32_PerfFormattedData_PerfOS_Memory get AvailableBytes /value^|findstr "AvailableBytes"') do (set freeMem=%%a)
 echo {"cpu_process": "%uCPU%","mem_free":"%freeMem%","mem_total":"%totalMem%"}
 ```
