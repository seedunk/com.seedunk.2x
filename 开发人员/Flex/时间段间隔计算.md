# 时间段间隔计算

```js
   private function getTimeUse(stime:String,etime:String):String{
var start:Array=stime.split(":");
var end:Array=etime.split(":");


var ss:int= int(start[2]);
var es:int= int(end[2]);
if(es<ss){
es=es+60;
if(end[1]==0){
if(end[0]==0){end[0]="23";}
else{end[0]=(int(end[0])-1).toString();}
end[1]="59";
}
else{end[1]=(int(end[1])-1).toString();}
}
var s:String=(es-ss).toString();

var sm:int= int(start[1]);
var em:int= int(end[1]);
if(em<sm){
em=em+60;
if(end[0]==0){end[0]="23";}
else{end[0]=(int(end[0])-1).toString();}
}

var m:String=(em-sm).toString();

var sh:int= int(start[0]);
var eh:int= int(end[0]);
if(eh<sh){eh=eh+24; }

var h:String=(eh-sh).toString();
return h+":"+m+":"+s;
}
```
